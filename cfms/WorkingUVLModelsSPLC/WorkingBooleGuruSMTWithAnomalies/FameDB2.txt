(set-logic QF_BV)
(define-fun implies ((p Bool) (q Bool)) Bool (or (not p) q))

(declare-const STANDARD_MODEL_NAME Bool)
(declare-const BTree Bool)
(declare-const Unindexed Bool)
(declare-const InMemory Bool)
(declare-const Persistent Bool)
(declare-const Dynamic Bool)
(declare-const Static Bool)
(declare-const LFU Bool)
(declare-const LRU Bool)
(declare-const NutOS Bool)
(declare-const Win Bool)
(declare-const DebugLogging Bool)
(declare-const delete Bool)
(declare-const get Bool)
(declare-const put Bool)
(assert
(and STANDARD_MODEL_NAME(and(implies(or BTree Unindexed)
 STANDARD_MODEL_NAME)
(implies STANDARD_MODEL_NAME(or(and BTree(not Unindexed))
(and(not BTree)
 Unindexed)))
(and(implies(or Dynamic Static)
 Persistent)
(implies Persistent(or(and Dynamic(not Static))
(and(not Dynamic)
 Static))))
(and(implies(or LFU LRU)
 Persistent)
(implies Persistent(or(and LFU(not LRU))
(and(not LFU)
 LRU))))
(and(implies(or InMemory Persistent)
 STANDARD_MODEL_NAME)
(implies STANDARD_MODEL_NAME(or(and InMemory(not Persistent))
(and(not InMemory)
 Persistent))))
(and(implies(or NutOS Win)
 STANDARD_MODEL_NAME)
(implies STANDARD_MODEL_NAME(or(and NutOS(not Win))
(and(not NutOS)
 Win))))
(implies DebugLogging STANDARD_MODEL_NAME)
(=(or delete get put)
 STANDARD_MODEL_NAME))
(and(implies NutOS BTree)
(implies Win BTree)))
