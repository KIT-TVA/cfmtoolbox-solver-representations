(set-logic QF_BV)
(define-fun implies ((p Bool) (q Bool)) Bool (or (not p) q))

(declare-const STANDARD_MODEL_NAME Bool)
(declare-const Basic Bool)
(declare-const Colour Bool)
(declare-const HighResolution Bool)
(declare-const GPS Bool)
(declare-const Media Bool)
(declare-const Camera Bool)
(declare-const MP3 Bool)
(assert
(and STANDARD_MODEL_NAME(and(implies(or Basic Colour HighResolution)
 STANDARD_MODEL_NAME)
(implies STANDARD_MODEL_NAME(or(and Basic(not Colour)
(not HighResolution))
(and(not Basic)
 Colour(not HighResolution))
(and(not Basic)
(not Colour)
 HighResolution)))
(=(or Camera MP3)
 Media)
(and(implies GPS STANDARD_MODEL_NAME)
(implies Media STANDARD_MODEL_NAME)))
(and(implies(not HighResolution)
(not Camera))
(implies GPS(not Basic))
(implies STANDARD_MODEL_NAME Media)))
