(set-logic QF_BV)
(define-fun implies ((p Bool) (q Bool)) Bool (or (not p) q))

(declare-const STANDARD_MODEL_NAME Bool)
(declare-const Gen_Plan_Analysis Bool)
(declare-const Lot Bool)
(declare-const Gen_Plan_Product Bool)
(declare-const Product Bool)
(declare-const Heat Bool)
(declare-const Sequence Bool)
(declare-const Cooling_L2 Bool)
(declare-const Cooling_Segment Bool)
(declare-const Gen_Watchdogs Bool)
(declare-const Mat_Tracking Bool)
(declare-const QA_C Bool)
(declare-const QA_A Bool)
(assert
(and STANDARD_MODEL_NAME(and(implies Gen_Plan_Product Lot)
(and(implies(or Gen_Plan_Analysis Lot Product)
 STANDARD_MODEL_NAME)
(implies STANDARD_MODEL_NAME(or(and Gen_Plan_Analysis(not Lot)
(not Product))
(and(not Gen_Plan_Analysis)
 Lot(not Product))
(and(not Gen_Plan_Analysis)
(not Lot)
 Product))))
(and(implies(or Heat Sequence)
 STANDARD_MODEL_NAME)
(implies STANDARD_MODEL_NAME(or(and Heat(not Sequence))
(and(not Heat)
 Sequence))))
(implies Cooling_Segment Cooling_L2)
(implies QA_A QA_C)
(and(implies Cooling_L2 STANDARD_MODEL_NAME)
(implies Gen_Watchdogs STANDARD_MODEL_NAME)
(implies Mat_Tracking STANDARD_MODEL_NAME)
(implies QA_C STANDARD_MODEL_NAME)))
(and(implies Heat Gen_Plan_Analysis)
(implies Sequence Product)))
